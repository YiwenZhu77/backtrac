% ============ Motivation ============
\begin{frame}{Motivation and Method}

\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Question}: What fraction of RC H$^+$ comes from bubbles?

\vspace{4pt}
\footnotesize
\begin{tabular}{@{}p{2.0cm}p{4.2cm}@{}}
\toprule
\textbf{Study} & \textbf{Finding} \\
\midrule
Sciola 2023 & $\geq$50\% RC energy from bubbles (enthalpy flux at 6\,R$_E$) \\
Yang 2015 & $\sim$60\% pressure from bubbles (RCM-E trace at 15\,R$_E$) \\
\bottomrule
\end{tabular}

\vspace{6pt}
\normalsize
\textbf{Our approach}: CHIMP full-orbit backward trace in GAMERA fields. Advantage over RCM: traces through bubble interiors with full Lorentz force.
\end{column}
\begin{column}{0.45\textwidth}
\footnotesize
\textbf{Pilot setup (same event as Sciola):}
\begin{itemize}
\item GAMERA-RCM, 17 March 2013 storm
\item 10{,}000 H$^+$, 10--200\,keV
\item Uniform in $r_\text{eq}=4.2$--6.6\,R$_E$
\item $\eta$ weights from RCM for analysis
\item Backward 4\,h: 10:00$\to$06:00\,UT
\item Fixed $\Delta t=0.01$\,s (Boris pusher)
\end{itemize}
\vspace{2pt}
\textbf{Bubble ID (Sciola):}\\
$\Delta B_\tau > 10$\,nT and $V_r<0$ on GAMERA equatorial plane arc, $\tau=2$\,min
\end{column}
\end{columns}
\end{frame}

% ============ Reversibility ============
\begin{frame}{Validation: Backward Trace Accuracy}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\centering
\includegraphics[width=\textwidth]{figures/reversibility_summary.png}

\vspace{2pt}
\footnotesize
\textbf{Boris pusher reversibility test:}\\
Forward then backward with same fixed $\Delta t$.
Roundtrip error $\sim 10^{-12}$\,R$_E$ (machine precision). Fixed timestep ensures exact time-reversal symmetry.
\end{column}
\begin{column}{0.50\textwidth}
\footnotesize
\textbf{Accuracy argument for FO backward trace:}
\begin{enumerate}
\item Boris pusher is symplectic $\to$ machine-precision reversible with fixed $\Delta t$
\item Drift decomposition confirms ExB + grad-curv explains 92\% of FO eq-plane motion (direction cos\,$\theta$ = 0.94)
\item H$^-$ charge reversal gives mirror-image dawn-dusk pattern, confirming grad-curv drift physics
\item Energy conservation: $|\Delta K|/K < 1\%$ over 4\,h trace
\end{enumerate}

\vspace{4pt}
\textbf{Conclusion:} FO backward trace faithfully captures the physical drift motion. The Boris pusher preserves the Hamiltonian structure, and the drift analysis independently confirms the trajectories are governed by the expected ExB + grad-curv physics.
\end{column}
\end{columns}

\item Bubble + pre-existing $\sim$54\% of RC at 15\,R$_E$, consistent with Sciola and Yang
\item H$^-$ reversal confirms dawn-dusk asymmetry is physical (grad-curv drift)
\end{itemize}
