% ============ Motivation ============
\begin{frame}{Motivation and Method}

\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Question}: What fraction of RC H$^+$ comes from bubbles?

\vspace{4pt}
\footnotesize
\begin{tabular}{@{}p{2.0cm}p{4.2cm}@{}}
\toprule
\textbf{Study} & \textbf{Finding} \\
\midrule
Sciola 2023 & $\geq$50\% RC energy from bubbles \\
Yang 2015 & $\sim$60\% pressure from bubbles \\
\bottomrule
\end{tabular}

\vspace{6pt}
\normalsize
\textbf{Two approaches compared}:
\begin{enumerate}
\item CHIMP full-orbit (FO) backward trace in GAMERA fields ($\mathbf{E}=-\mathbf{V}\times\mathbf{B}$, Boris pusher)
\item Sina's RCM backtracker ($V_\text{eff} = \Phi + \lambda V_M$, bounce-averaged drift)
\end{enumerate}
\end{column}
\begin{column}{0.45\textwidth}
\footnotesize
\textbf{Pilot setup}:
\begin{itemize}
\item GAMERA-RCM (sp13\_075), 17 March 2013
\item 1{,}000 H$^+$, 10--200\,keV
\item Uniform in $r_\text{eq}=4.2$--6.6\,R$_E$
\item Backward 4\,h: 10:00$\to$06:00\,UT
\item FO: fixed $\Delta t=0.01$\,s, $x_\text{eq}/y_\text{eq}$ from field line trace
\item Sina: $V_\text{eff}$ drift on $(I,J)$ grid, RK4 with 10 substeps/min
\end{itemize}
\vspace{2pt}
\textbf{Bubble criteria}:\\
$\Delta B_z > 15$\,nT on GAMERA eq.\ plane,\\
midnight sector (MLT 21--03),\\
$x<0$, $r>6$\,R$_E$
\end{column}
\end{columns}
\end{frame}

% ============ Sina adaptation ============
\begin{frame}{Sina's RCM Backtracker: Adaptation for MAGE}

\begin{columns}[T]
\begin{column}{0.50\textwidth}
\textbf{Sina's method} (Sadeghzadeh, standalone RCM):
\begin{itemize}
\item Tracks $\eta$ channels in $(I,J)$ ionospheric grid
\item $V_\text{eff} = V + \lambda \cdot V_M$
\item $\lambda$: adiabatic invariant ($K = |\lambda| \cdot V_M / 1000$)
\item Inductive $\mathbf{E}$ implicit via $X_\text{MIN}(I,J,t)$ evolution
\item Dipole $B_r$ in Jacobian ($\sim$4\% error vs actual)
\end{itemize}

\vspace{4pt}
\textbf{Key fix for MAGE output}:\\
MAGE \texttt{colat} stored in \textbf{radians}; Sina's code assumed degrees. Double conversion shrunk $\sin\theta$ by $57\times$ $\to$ \texttt{fac} too small $\to$ 80\% particle death in 5 steps. Fixed all RK4 stages (k1--k4).
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Validation}:

\vspace{2pt}
\footnotesize
\begin{tabular}{@{}lrr@{}}
\toprule
 & Sina fixed & Sina rewrite \\
\midrule
$\cos\theta$ (5\,min) & 0.981 & 0.982 \\
$\cos\theta$ (20\,min) & 0.880 & 0.880 \\
err (5\,min) & 0.59\,R$_E$ & 0.59\,R$_E$ \\
Survive (60\,min) & 744 & 896 \\
\bottomrule
\end{tabular}
\normalsize

\vspace{6pt}
After fix, Sina's original code and our rewrite give identical trajectories. Survival difference from ellipse masking in original \texttt{data\_init.py}.

\vspace{4pt}
Code: \texttt{backtrac/rcm\_backtrac/} (git)\\
\texttt{backtrac/pilot/run\_sina\_1k.py}
\end{column}
\end{columns}
\end{frame}

% ============ Bubble results ============
\begin{frame}{Results: Bubble Contribution to Ring Current}

\begin{columns}[T]
\begin{column}{0.50\textwidth}
\footnotesize
\textbf{Origin classification} (ever-in-bubble criterion):\\
$\Delta B_z > 15$\,nT, midnight MLT 21--03, $x<0$, $r>6$\,R$_E$

\vspace{4pt}
\begin{tabular}{@{}lrr@{}}
\toprule
 & \textbf{FO} & \textbf{Sina} \\
\midrule
Bubble & 30\% & 16\% \\
Pre-existing & 15\% & 70\% \\
Convection & 56\% & 15\% \\
\bottomrule
\end{tabular}

\vspace{6pt}
\normalsize
\textbf{By energy}:

\vspace{2pt}
\footnotesize
\begin{tabular}{@{}lrr@{}}
\toprule
Energy & FO bubble & Sina bubble \\
\midrule
10--30\,keV & 52\% & 26\% \\
30--60\,keV & 38\% & 27\% \\
60--120\,keV & 23\% & 6\% \\
120--200\,keV & 6\% & 6\% \\
\bottomrule
\end{tabular}
\end{column}
\begin{column}{0.48\textwidth}
\footnotesize
\textbf{Key findings}:
\begin{itemize}
\item FO sees \textbf{more bubbles} (30\% vs 16\%) because full-orbit gc drift carries particles through bubble channels faster
\item Sina has \textbf{more pre-existing} (70\% vs 15\%) because bounce-averaged drift is slower $\to$ particles don't escape RC in 4\,h
\item Low-energy particles ($<$30\,keV) have highest bubble fraction in both methods --- ExB convection efficiently transports them through bubbles
\item High-energy ($>$120\,keV) bubble fraction converges ($\sim$6\%) --- these particles drift azimuthally too fast to be captured by bubbles
\end{itemize}

\vspace{4pt}
\normalsize
\textbf{FO is more trustworthy}: complete Lorentz force, includes gc drift and inductive E in $\mathbf{E}=-\mathbf{V}\times\mathbf{B}$
\end{column}
\end{columns}
\end{frame}

% ============ RCM domain ============
\begin{frame}{RCM Domain Structure in MAGE}

\begin{columns}[T]
\begin{column}{0.50\textwidth}
\centering
\includegraphics[width=\textwidth]{figures/rcm_boundary_and_jumpers.png}
\end{column}
\begin{column}{0.48\textwidth}
\footnotesize
\textbf{Three zones on RCM grid}:
\begin{enumerate}
\item \texttt{CLOSED} ($i_\text{open}=-1$): RCM active, $\eta$ advected by $V_\text{eff}$
\item \texttt{NULL} ($i_\text{open}=0$): closed field lines but outside \texttt{bndloc} --- buffer zone, VM valid but $\eta$ not pushed
\item \texttt{OPEN} ($i_\text{open}=+1$): open field lines
\end{enumerate}

\vspace{4pt}
\textbf{Boundary (\texttt{bndloc})} determined by:
\begin{itemize}
\item Open/closed from GAMERA field line trace
\item Smoothed in longitude ($\sim$1\,hr MLT)
\item ``Only shrink'' constraint $\to$ dayside compression pulls entire boundary earthward
\end{itemize}

\vspace{4pt}
Nightside: $\sim$10--15\,R$_E$\\
Dayside: $\sim$5--6\,R$_E$ (storm compression)\\
Time-varying (updated each coupling step)
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Discussion and Next Steps}

\begin{columns}[T]
\begin{column}{0.50\textwidth}
\textbf{Summary}:
\begin{itemize}
\item FO backward trace: 30\% bubble origin
\item Sina $V_\text{eff}$: 16\% bubble (slower drift $\to$ fewer reach tail)
\item Both agree bubbles concentrate in 10--60\,keV range
\item Drift decomposition validates FO: ExB+gc explains 92\% of equatorial drift
\item Boris pusher machine-precision reversible with fixed $\Delta t$
\end{itemize}

\vspace{4pt}
\textbf{Sina code adaptation}:
\begin{itemize}
\item MAGE \texttt{colat} in radians (not degrees)
\item $\lambda = 1000 K / V_M$ (positive for H$^+$)
\item Codex audit: found k2/k3/k4 colat bug
\end{itemize}
\end{column}
