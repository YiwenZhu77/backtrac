% ============ Motivation ============
\begin{frame}{Motivation and Method}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Question}: What fraction of ring current H$^+$ originates from plasma sheet bubbles?

\vspace{6pt}
\footnotesize
\begin{tabular}{@{}p{1.8cm}p{3.8cm}@{}}
\toprule
\textbf{Study} & \textbf{Finding} \\
\midrule
Sciola 2023 & $\geq$50\% RC energy from bubbles \\
Yang 2015 & $\sim$60\% pressure from bubbles \\
\bottomrule
\end{tabular}

\vspace{6pt}
\normalsize
\textbf{Two approaches}:
\begin{enumerate}
\item \textbf{FO}: CHIMP full-orbit backward trace\\
{\footnotesize Boris pusher, $\mathbf{E}=-\mathbf{V}\times\mathbf{B}$ from GAMERA}
\item \textbf{Sina's}: RCM backtracker\\
{\footnotesize $V_\text{eff} = \Phi + \lambda V_M$, bounce-averaged drift}
\end{enumerate}
\end{column}
\begin{column}{0.50\textwidth}
\footnotesize
\textbf{Pilot setup}:
\begin{itemize}\setlength\itemsep{1pt}
\item GAMERA-RCM (sp13\_075), 17 March 2013
\item 1{,}000 H$^+$, 10--200\,keV, $r_\text{eq}=4.2$--6.6\,R$_E$
\item Backward 4\,h: 10:00$\to$06:00\,UT
\item FO: fixed $\Delta t=0.01$\,s, equatorial projection via field line trace
\item Sina: $V_\text{eff}$ drift on $(I,J)$ ionospheric grid
\end{itemize}

\vspace{4pt}
\textbf{Bubble identification}:

A particle is \textit{bubble origin} if it \textbf{ever} passes through a region satisfying:

\vspace{2pt}
\begin{tabular}{@{}ll@{}}
$\Delta B_z > 15$\,nT & (dipolarization) \\
MLT 21--03 & (midnight sector) \\
$x < 0$, $r > 6$\,R$_E$ & (nightside, not inner RC) \\
\end{tabular}
\end{column}
\end{columns}
\end{frame}

% ============ Bubble results ============
\begin{frame}{Results: Bubble Contribution to Ring Current}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\footnotesize
\textbf{Origin classification}:

\vspace{4pt}
\begin{tabular}{@{}lrr@{}}
\toprule
 & \textbf{FO} & \textbf{Sina} \\
\midrule
Bubble & 30\% & 16\% \\
Pre-existing & 15\% & 70\% \\
Convection & 56\% & 15\% \\
\bottomrule
\end{tabular}

\vspace{6pt}
\textbf{By energy}:

\vspace{2pt}
\begin{tabular}{@{}lrr@{}}
\toprule
Energy & FO bubble & Sina bubble \\
\midrule
10--30\,keV & 52\% & 26\% \\
30--60\,keV & 38\% & 27\% \\
60--120\,keV & 23\% & 6\% \\
120--200\,keV & 6\% & 6\% \\
\bottomrule
\end{tabular}
\end{column}
\begin{column}{0.50\textwidth}
\footnotesize
\textbf{Key findings}:
\begin{itemize}\setlength\itemsep{2pt}
\item FO sees more bubbles (30\% vs 16\%): full-orbit grad-curv drift carries particles through bubble channels faster than bounce-averaged drift
\item Sina has more pre-existing (70\% vs 15\%): slower drift means particles don't escape RC in 4\,h
\item Low-energy ($<$30\,keV): highest bubble fraction --- ExB convection efficiently transports through bubbles
\item High-energy ($>$120\,keV): bubble fraction converges ($\sim$6\%) --- fast azimuthal drift outruns bubbles
\item FO direction agrees with Sina at $\cos\theta = 0.98$ (5\,min); speed ratio $\approx 0.83$
\end{itemize}
\end{column}
\end{columns}
\end{frame}
